##1.视频相关知识的简介

1.1封装格式

####@封装格式的主要作用是把视频码流和音频码流按照一定的格式存储在一个文件中

####@也就是说 一个MP4的封装格式的一个文件，它的视频编码方式可以是MPEG-2, MPEG-4, H.264, H.263其中任意一种

1.2视频编码的标准：

####@rgb, yuv 这些是像素的数据格式，把像素的数据压缩成视频码流的过程就是视频编码。

####@如果这些像素数据不经过压缩，10秒的数据就有1G。

1.3 视频压缩

刚才说了，如果原始视频不经过压缩，就会很大。那压缩从另一个角度分，分有损压缩和无损压缩。常用的是有损压缩，牺牲画面的质量。比如，如果视频一个区域都是红色，那么就不需要在这个区域所有的点都标致为红色，只要把轮廓标致为红色，就可以了。这就是空间压缩。还有时间压缩。视频压缩（对某些信息不敏感就忽略或者不采集很多zhen画面）。

1.4 视频播放

视频播放其实是在解码，刚才上面说的都是视频编码。绝大部分的播放器都是把不同的封装格式的视频转为同一种格式来播放，比如yuv。（而音频都是转成pcm播放）

一个视频文件包含视频流和音频流和字幕流。

##2.ffmpeg 对视频的处理

###2.1 AVFormatContext 视频文件上下文

AVFormatContext是一个贯穿始终的数据结构，很多函数都要用到它作为参数。它是FFMPEG解封装（flv，mp4，rmvb，avi）功能的结构体。下面看几个主要变量的作用（在这里考虑解码的情况）：

AVIOContext *pb：输入数据的缓存
unsigned int nb_streams：视音频流的个数
AVStream **streams：视音频流
char filename[1024]：文件名
int64_t duration：时长（单位：微秒us，转换为秒需要除以1000000）
int bit_rate：比特率（单位bps，转换为kbps需要除以1000）
AVDictionary *metadata：元数据

以上内容摘自FFMPEG结构体分析：AVFormatContext

const char inputStr = env->GetStringUTFChars(inputStr_, 0);
const char outStr = env->GetStringUTFChars(outStr_, 0);
//注册各大组件
av_register_all();
//获取av的上下文
AVFormatContext *pContext = avformat_alloc_context();

//根据地址打开文件
if(avformat_open_input(&pContext, inputStr, NULL, NULL)<0) {
    LOGE("打开失败");
    return;
}
//获取视频信息
if (avformat_find_stream_info(pContext, NULL) < 0) {
    LOGE("获取信息失败");
    return;
}

int vedio_stream_idx=-1;
//找到视频流pContext->nb_streams 这里包含所有码流  音频 视频 字幕
for (int i = 0; i < pContext->nb_streams; ++i) {
    LOGE("循环  %d", i);
    //codec 每一个流 对应的解码上下文   codec_type 流的类型
    if (pContext->streams[i]->codec->codec_type == VMEDIA_TYPE_VIDEO)           {
        vedio_stream_idx = i;
    }
}

//获取到解码器上下文
AVCodecContext* pCodecCtx=pContext->streams[vedio_stream_idx]->codec;

//解码器
AVCodec *pCodex=avcodec_find_decoder(pCodecCtx->codec_id);
//ffempg版本升级
if (avcodec_open2(pCodecCtx, pCodex, NULL) < 0) {
    LOGE("解码失败");
    return;
}
//    分配内存   malloc  AVPacket   1   2
AVPacket *packet = (AVPacket *) av_malloc(sizeof(AVPacket));
//    初始化结构体
av_init_packet(packet);
//还是不够
AVFrame *frame = av_frame_alloc();
//    声明一个yuvframe
AVFrame *yuvFrame = av_frame_alloc();
//    给yuvframe  的缓冲区 初始化

uint8_t  *out_buffer= (uint8_t *)        av_malloc(avpicture_get_size(AV_PIX_FMT_YUV420P, pCodecCtx->width, pCodecCtx->height));

int re=avpicture_fill((AVPicture *) yuvFrame, out_buffer, AV_PIX_FMT_YUV420P, pCodecCtx->width, pCodecCtx->height);
LOGE("宽 %d  高 %d",pCodecCtx->width,pCodecCtx->height);
//    mp4   的上下文pCodecCtx->pix_fmt
SwsContext *swsContext=sws_getContext(pCodecCtx->width,pCodecCtx->height,pCodecCtx->pix_fmt,
                                      pCodecCtx->width,pCodecCtx->height,AV_PIX_FMT_YUV420P
        ,SWS_BILINEAR,NULL,NULL,NULL
);
int frameCount = 0;
FILE *fp_yuv = fopen(outStr, "wb");

//packet入参 出参对象  转换上下文
int got_frame;
while (av_read_frame(pContext, packet) >= 0) {
//        节封装

//        根据frame 进行原生绘制    bitmap  window
    avcodec_decode_video2(pCodecCtx, frame, &got_frame, packet);
//   frame  的数据拿到   视频像素数据 yuv   三个rgb    r   g  b   数据量大   三个通道
//        r  g  b  1824年    yuv 1970
    LOGE("解码%d  ",frameCount++);
    if (got_frame > 0) {
        sws_scale(swsContext, (const uint8_t *const *) frame->data, frame->linesize, 0, frame->height, yuvFrame->data,
                  yuvFrame->linesize
        );
        int y_size = pCodecCtx->width * pCodecCtx->height;
//        y 亮度信息写完了
        fwrite(yuvFrame->data[0], 1, y_size, fp_yuv);
        fwrite(yuvFrame->data[1], 1, y_size/4, fp_yuv);
        fwrite(yuvFrame->data[2], 1, y_size/4, fp_yuv);
    }
    av_free_packet(packet);
}
fclose(fp_yuv);
av_frame_free(&frame);
av_frame_free(&yuvFrame);
avcodec_close(pCodecCtx);
avformat_free_context(pContext);
env->ReleaseStringUTFChars(inputStr_, inputStr);
env->ReleaseStringUTFChars(outStr_, outStr);
