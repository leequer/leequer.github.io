插入排序以及改进版

发表于 2013-01-27   |    阅读次数

插入排序可以提前终止。

目标：
一般来说，插入排序都采用在数组上实现。具体算法描述如下：
⒈ 从第一个元素开始，该元素可以认为已经被排序
⒉ 取出下一个元素，在已经排序的元素序列中从后向前扫描
⒊ 如果该元素（已排序）大于新元素，将该元素和新元素互换
⒋ 重复步骤3，直到找到已排序的元素小于或者等于新元素的位置
⒌ 将新元素插入到下一位置中
⒍ 重复步骤2~5

import java.util.Random;

public class InsertSort {

    public static void insertSort() {

    }

    public static int[] makeArray() {
        int size = 10000;
        int a[] = new int[size];
        for (int i = 0; i < size; i++) {
            a[i] = (new Random().nextInt()) % (size - 0 + 1);
        }
        return a;
    }

    public static void main(String[] args) {
        int[] array = makeArray();
        int size = array.length;

        long startTime = System.currentTimeMillis(); // 获取开始时间
        for (int i = 1; i < size; i++) {
            int j = i;

            for (; j > 0; j--) {
                int left = array[j - 1];
                int right = array[j];
                if (left > right) {

                    int temp = array[j - 1];
                    array[j - 1] = right;
                    array[j] = temp;
                } else {
                    break;
                }
            }

        }
        long endTime = System.currentTimeMillis(); // 获取结束时间

        System.out.println("程序运行时间：" + (endTime - startTime) + "ms"); // 输出程序运行时间
        for (int i = 0; i < size; i++) {
            System.out.println(array[i]);
        }

    }

}

/**
 * 插入排序，并没有选择排序快，因为每次交换都要有三次赋值
 * （int temp = array[j-1];
    array[j-1]= right;
    array[j] = temp;），

 * 此优化版本思路，先准备当前位置的副本，每次比较后，直接赋值，最后把副本赋值到应该的位置
 * 
 * 对于源数据基本上是有序排列的时候，这种插入排序的效率会
 * @author liqing
 *
 */

        int a [] = new int[size] ;
        for (int i = 0; i <size; i++) {

        }
        return a;
    }

    public static void main(String[] args) {
          int [] array = makeArray();

        long startTime = System.currentTimeMillis(); // 获取开始时间

// 这是自己的写算法，下面是视频上写的算法，速度是不一样的

            for (int j = i; j > 0; j--) {
