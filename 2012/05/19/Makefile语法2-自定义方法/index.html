Makefile语法2-自定义方法，递归式展开

#1自定义方法

#自定义方法myfun 两个参数 
myfun=$2 $1
#自定义方法myfun_ret 调用myfun方法 传入第二个参数20 第一个参数10 注意顺序 是反着的
myfun_ret=$(call myfun,20,10)
#输出
test:
    @echo $(myfun_ret)

###$>执行make test 结果是10 20

#2递归式展开 是用等号 =

str2是str1赋值的，递归展开寻找

str2=$(str1)
str1=hello
test:
    @echo $(str2)

###$>执行make 或者 make test 结果是hello

str2=$(str1)
str1=hello
str1=world
test:
    @echo $(str2)

###$>这个会输出什么呢？ 结果是world

#3直接式展开 是分号等号 ：=

str1:=android
str1:=ios
str2:=$（str1）
test:
    @echo $(str2)

###$>输出结果 ios
